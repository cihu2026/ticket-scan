* 正數為增加，負數為減少

// ⚠️ 請確保此 URL 與你最新部署的 GAS 網址一致 const API_URL = "https://script.google.com/macros/s/AKfycbw5VQi-dtLodRQbKTgzueGGXgMBBedA_Th93Skr_Dia3-JYbxtd00shnZ24axbznDY/exec"; let html5QrCode; let scanning = false; let lastTicketId = ""; let adjusted = false; const statusEl = document.getElementById("status"); const adjustArea = document.getElementById("adjustArea"); // 啟動相機 document.getElementById("start").onclick = () => { if (html5QrCode) { html5QrCode.stop().then(() => startCamera()); } else { startCamera(); } }; function startCamera() { html5QrCode = new Html5Qrcode("reader"); html5QrCode.start( { facingMode: "environment" }, { fps: 10, qrbox: { width: 250, height: 250 } }, onScan ).catch(err => { statusEl.className = "status err"; statusEl.innerText = "❌ 無法啟動相機"; }); statusEl.className = "status wait"; statusEl.innerText = "📷 相機掃描中..."; document.getElementById("start").style.display = "none"; } // 掃描到 QR Code 時的動作 function onScan(text) { if (scanning) return; scanning = true; // 解析 ID (支援完整網址或純 ID) const id = text.includes("id=") ? text.split("id=")[1].split("&")[0] : text; lastTicketId = id; adjusted = false; adjustArea.style.display = "none"; statusEl.className = "status wait"; statusEl.innerText = "⏳ 正在驗票中..."; fetch(API_URL, { method: "POST", body: JSON.stringify({ id: id }) }) .then(r => r.json()) .then(res => { if (!res.ok) { statusEl.className = "status err"; statusEl.innerText = res.msg; // 2秒後允許再次掃描 setTimeout(() => { scanning = false; statusEl.innerText = "📷 繼續掃描..."; }, 2000); return; } const d = res.detail; statusEl.className = "status ok"; statusEl.innerText = `✅ 驗票成功！已自動入園\n👤 總人數：${d.people}\n(餘額：場內目前 ${res.current || '?'} 人)`; // 顯示微調按鈕 adjustArea.style.display = "block"; // 5秒後自動隱藏微調區並允許再次掃描 setTimeout(() => { scanning = false; statusEl.className = "status wait"; statusEl.innerText = "📷 繼續掃描..."; }, 5000); }) .catch((err) => { statusEl.className = "status err"; statusEl.innerText = "❌ 網路連線失敗"; scanning = false; }); } // 手動微調功能 function adjust(diff) { if (adjusted) return; if (!confirm(`確認手動微調 ${diff > 0 ? "+" : ""}${diff} 人？`)) return; fetch(API_URL, { method: "POST", body: JSON.stringify({ type: "IN", delta: diff, source: "驗票端微調", note: "票號:" + lastTicketId }) }) .then(() => { adjusted = true; adjustArea.style.display = "none"; statusEl.innerText += `\n(已微調 ${diff} 人)`; }); }
